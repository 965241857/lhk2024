课题分类颜色标识设置
用于设置不同类型课题的标识颜色，便于在选题列表中快速区分课题类型
系统预设了常用课题类型，您可以为每种类型设置独特的标识颜色
用于标识软件开发类课题
用于标识理论研究类课题
用于标识实验分析类课题
用于标识设计类课题
用于标识调研报告类课题
    $(function () {
        // 软件开发类课题颜色选择器
        $('.test-select-color').paigusu({
            color: '#1aa094',//初始色
        }, function (event, obj) {
            $(event).css('background-color', '#' + obj.hex);
            $('input[name="software_color"]').val('#' + obj.hex);
        });
        
        // 理论研究类课题颜色选择器
        $('.test-select-color2').paigusu({
            color: '#3398db',//初始色
        }, function (event, obj) {
            $(event).css('background-color', '#' + obj.hex);
            $('input[name="theory_color"]').val('#' + obj.hex);
        });
        
        // 更多类型的颜色选择器配置...
        
        // 表单提交处理
        layui.use('form', function() {
            var form = layui.form;
            form.on('submit(saveColorConfig)', function(data) {
                // 保存颜色配置到服务器
                console.log('颜色配置数据:', data.field);
                layer.msg('颜色配置保存成功', {icon: 1});
                return false;
            });
        });
    });
